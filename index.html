⚠ Memerlukan update GAS. Tambahkan kode di bawah ke Apps Script yang sama, lalu redeploy.
Tambahkan fungsi ini ke GAS (tanpa menghapus doPost yang lama):
function doGet(e) { try { const p = e.parameter; if (p.action === 'list') return listFiles(p.folderId); if (p.action === 'read') return readFile(p.fileId); return out({status:'error', msg:'unknown action'}); } catch(err) { return out({status:'error', msg:err.toString()}); } } function listFiles(folderId) { const folder = folderId ? DriveApp.getFolderById(folderId) : DriveApp.getRootFolder(); const iter = folder.getFiles(), files = []; while (iter.hasNext()) { const f = iter.next(); if (f.getName().endsWith('.json')) files.push({id:f.getId(), name:f.getName()}); } return out({status:'ok', files}); } function readFile(fileId) { const f = DriveApp.getFileById(fileId); return ContentService.createTextOutput(f.getBlob().getDataAsString()) .setMimeType(ContentService.MimeType.JSON); }
