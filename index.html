function doGet(e) {
  try {
    const action = e.parameter.action;
    const folderId = e.parameter.folderId;
    if (action === 'list') return listFiles(folderId);
    if (action === 'read') return readFile(e.parameter.fileId);
    return out({status:'error',msg:'unknown action'});
  } catch(err) { return out({status:'error',msg:err.toString()}); }
}

function listFiles(folderId) {
  const folder = folderId
    ? DriveApp.getFolderById(folderId)
    : DriveApp.getRootFolder();
  const iter = folder.getFiles();
  const files = [];
  while (iter.hasNext()) {
    const f = iter.next();
    if (f.getName().endsWith('.json'))
      files.push({id:f.getId(),name:f.getName(),
        date:f.getDateCreated().toISOString()});
  }
  return out({status:'ok', files});
}

function readFile(fileId) {
  const f = DriveApp.getFileById(fileId);
  return ContentService
    .createTextOutput(f.getBlob().getDataAsString())
    .setMimeType(ContentService.MimeType.JSON);
}
