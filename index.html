⚠ Memerlukan update GAS. Tambahkan kode di bawah ke Apps Script yang sama, lalu redeploy.
Tambahkan fungsi ini ke GAS (tanpa menghapus doPost yang lama):
// Tambahkan semua fungsi ini ke Apps Script (jangan hapus doPost lama) function out(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { try { const p = e.parameter; if (p.action === 'list') return listFiles(p.folderId); if (p.action === 'read') return readFile(p.fileId); return out({status:'error', msg:'unknown action'}); } catch(err) { return out({status:'error', msg:err.toString()}); } } function listFiles(folderId) { const folder = folderId ? DriveApp.getFolderById(folderId) : DriveApp.getRootFolder(); const iter = folder.getFiles(), files = []; while (iter.hasNext()) { const f = iter.next(); if (f.getName().endsWith('.json')) files.push({id: f.getId(), name: f.getName()}); } return out({status:'ok', files}); } function readFile(fileId) { try { const f = DriveApp.getFileById(fileId); const data = JSON.parse(f.getBlob().getDataAsString()); return out({status:'ok', data: data}); } catch(err) { return out({status:'error', msg: err.toString()}); } }
ℹ️ Upload file .json yang sudah disimpan dari tombol → Sheets di WasteHub. Bisa upload banyak sekaligus.
